## 1. Zip two arrays

Assumptions:
1. Two arrays of strings.
2. Zip till the shorter one.

```typescript
function zipArray1(a: Array<string>, b: Array<string>): Array<string> {
    const la = a.length;
    const lb = b.length;
    const len = la < lb ? la : lb;
    const result: Array<string> = [];
    for (let i = 0; i < len; i++) {
        result[i] = a[i] + b[i];
    }
    return result;
}

const a1 = [1, 2, 3, 4, 5].map(x => x.toString());
const b1 = [0, 9, 8, 7, 6, 5, 4].map(x => x.toString());
console.log(zipArray1(a1, b1));
// ["10", "29", "38", "47", "56"] 
```


## 2. Zip two strings

Assumpions:
1. Two strings
2. Zip till the shorter one

```typescript
function zipString1(a: string, b: string): string {
    const la = a.length;
    const lb = b.length;
    const len = la < lb ? la : lb;
    const result: Array<string> = [];
    for (let i = 0; i < len; i++) {
        result[i] = a[i] + b[i];
    }
    return result.join('');
}
const a2 = [1, 2, 3, 4, 5].map(x => x.toString()).join('');
const b2 = [0, 9, 8, 7, 6, 5, 4].map(x => x.toString()).join('');
console.log(zipString1(a2, b2));
// "1029384756"
```

## 3. Zip N Arrays

Assumptions:
1. Multiple arrays passed down in first param
2. param contains array of strings
3. Length of output is that of the shortest array

```typescript
function ZipNArrays(as: Array<Array<string>>): Array<string> {
    if(as.length === 0) {
        return [];
    }
    const len = Math.min(...as.map(v => v.length));
    const result : Array<string> = new Array<string>(len);
    for(let i = 0; i < len; i++) {
        result[i] = as.map(a => a[i]).join('');
    }
    return result;
}
```
